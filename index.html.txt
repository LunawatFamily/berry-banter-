// --- Core Application Logic ---

/** Toggles the visibility of a recipe's detailed section. */
function toggleDetails(recipeId) {
const detailsElement = document.getElementById(`recipe-details-${recipeId}`);
const buttonElement = document.getElementById(`toggle-button-${recipeId}`);

if (detailsElement) {
const isCollapsed = detailsElement.classList.contains('open');

// Toggle the 'open' class for the CSS transition
detailsElement.classList.toggle('open');

// Update the button icon and rotation
if (buttonElement) {
const iconSvg = isCollapsed ? getIcon('ChevronDown', 24) : getIcon('ChevronUp', 24);
buttonElement.innerHTML = iconSvg;
// Removed the rotate-180 class toggle as it's not strictly necessary for chevron up/down
buttonElement.setAttribute('aria-expanded', !isCollapsed);
buttonElement.title = isCollapsed ? "Collapse Details" : "Expand Details";
}
}
}

/** Filters recipes based on the search term. */
function filterRecipes(searchTerm) {
const lowerCaseSearch = searchTerm.toLowerCase();

return initialRecipes.filter(recipe => {
// 1. Check title
if (recipe.title.toLowerCase().includes(lowerCaseSearch)) return true;

// 2. Check description
if (recipe.description.toLowerCase().includes(lowerCaseSearch)) return true;

// 3. Check ingredients
const ingredientMatch = recipe.ingredients.some(ingredient =>
ingredient.toLowerCase().includes(lowerCaseSearch)
);
if (ingredientMatch) return true;

return false;
});
}

/** Main rendering function to update the DOM. */
function renderRecipes() {
const listContainer = document.getElementById('recipe-list');
const searchTerm = document.getElementById('search-input').value;
const filtered = filterRecipes(searchTerm);

listContainer.innerHTML = ''; // Clear existing content

if (filtered.length > 0) {
filtered.forEach(recipe => {
listContainer.appendChild(renderRecipeCard(recipe));
});
} else {
// Display 'No results found' message
listContainer.innerHTML = `
<div class="md:col-span-full text-center p-16 bg-white/70 backdrop-blur-sm rounded-xl shadow-lg border border-dashed border-blue-300 text-blue-700">
<p class="text-3xl font-serif text-blue-800 mb-4">
Oops! No recipes found for "${searchTerm}"
</p>
<p class="text-lg font-light">Try adjusting your search or explore other delightful dishes.</p>
</div>
`;
}
}
